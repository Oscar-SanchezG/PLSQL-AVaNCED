DECLARE

  TYPE EMPLEADOS IS TABLE OF EMPLOYEES%ROWTYPE INDEX BY PLS_INTEGER;
 
  EMPLES EMPLEADOS;

BEGIN
     SELECT * 
     BULK COLLECT INTO EMPLES
     FROM EMPLOYEES WHERE SALARY > 5000;
    
     FOR I IN 1..EMPLES.COUNT() LOOP
        DBMS_OUTPUT.PUT_LINE(EMPLES(I).FIRST_NAME);
     END LOOP;
END;
